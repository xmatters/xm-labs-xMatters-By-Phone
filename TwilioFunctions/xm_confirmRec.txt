  let twiml = new Twilio.twiml.VoiceResponse();
    
    // Recording is good

            twiml.say({ voice: 'alice', loop: 1},'Hang on a moment while I transcribe your audio.');

            twiml.say({ voice: 'alice', loop: 1},'Without failure, there is no achievement. Lets try again.');
